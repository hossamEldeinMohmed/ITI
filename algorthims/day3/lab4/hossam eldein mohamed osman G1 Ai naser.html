nodes = {
    0: [1, 2],
    1: [0, 5, 3, 6],
    2: [0, 5, 6],
    3: [1, 7],
    4: [7, 6],
    5: [1, 2, 7],

G = nx.Graph(nodes)

G.add_edge(0,1,weight=8)
G.add_edge(0,2,weight=9)
G.add_edge(1,3,weight=1)

labels = nx.get_edge_attributes(G,'weight')
nx.draw_networkx_edge_labels(G,pos,edge_labels=labels)
